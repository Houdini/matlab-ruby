    def initialize(m, n, *names)
      super(m, n)
      @names = names
      
      # Populate the matrix with a hash using the names provided
      m.times do |row_index|
        n.times do |column_index|
          @cells[row_index][column_index] = names.inject({}) { |s,e| s.merge( { e.to_s => nil } ) }
        end
      end
    end
